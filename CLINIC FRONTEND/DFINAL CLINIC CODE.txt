        const themeToggle = document.getElementById('themeToggle');
        const body = document.body;

        themeToggle.addEventListener('click', () => {
            body.classList.toggle('dark-mode');
            if (body.classList.contains('dark-mode')) {
                themeIcon.classList.replace('fa-moon', 'fa-sun');
            } else {
                themeIcon.classList.replace('fa-sun', 'fa-moon');

        document.querySelectorAll('.nav-link').forEach(anchor => {
            anchor.addEventListener('click', function(e) {
                e.preventDefault();
                const targetId = this.getAttribute('href');
                const targetSection = document.querySelector(targetId);
                const offset = 100; // Adjusted for sticky header height

                window.scrollTo({
                    top: targetSection.offsetTop - offset,
                    behavior: 'smooth'
                });
            });
        });

        // AI Image Pre-Check Tool Logic
        const uploadContainer = document.getElementById('uploadContainer');
        const imageUpload = document.getElementById('imageUpload');
        const imagePreview = document.getElementById('imagePreview');
        const analysisSection = document.getElementById('analysisSection');
        const aiResult = document.getElementById('aiResult');
        const aiText = document.getElementById('aiText');
        const dropZone = document.getElementById('dropZone');

        uploadContainer.addEventListener('click', () => imageUpload.click());

        imageUpload.addEventListener('change', function(e) {
            const file = e.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = function(event) {
                    imagePreview.src = event.target.result;
                    imagePreview.classList.remove('hidden');
                    dropZone.classList.add('hidden');
                   
                }
                reader.readAsDataURL(file);
            }
        });

        function simulateAnalysis() {
            analysisSection.classList.remove('hidden');
            aiResult.classList.add('hidden');

            setTimeout(() => {
                analysisSection.classList.add('hidden');
                aiResult.classList.remove('hidden');
                
                const suggestions = [
                    "Based on the visual indicators, there appears to be moderate gingival inflammation. We recommend a professional cleaning within 48 hours.",
                    "The white spots detected could indicate early enamel demineralization. An appointment for fluoride treatment is suggested.",
                    "Potential deep cavity detected. We recommend an urgent checkup to prevent nerve damage.",
                    "Signs of bruxism (teeth grinding) detected. We should discuss custom night guards during your next visit."
                ];
                
                aiText.innerText = suggestions[Math.floor(Math.random() * suggestions.length)];
            }, 2500);
        }

            imagePreview.classList.add('hidden');
            dropZone.classList.remove('hidden');
            aiResult.classList.add('hidden');
            imageUpload.value = '';
        }

        // Chatbot Widget Logic
        const chatTrigger = document.getElementById('chatbot-trigger');
        const chatWindow = document.getElementById('chatbot-window');
        const closeChat = document.getElementById('close-chatbot');
        const chatInput = document.getElementById('chat-input');
        const sendChat = document.getElementById('send-chat');
        const chatMessages = document.getElementById('chat-messages');

        chatTrigger.addEventListener('click', () => {
            chatWindow.style.display = chatWindow.style.display === 'flex' ? 'none' : 'flex';

        closeChat.addEventListener('click', () => {
            chatWindow.style.display = 'none';
        });

        function addMessage(text, sender) {
            const div = document.createElement('div');
            div.className = sender === 'user' 
                ? 'bg-teal-600 text-white p-3 rounded-lg max-w-[80%] self-end ml-auto' 
                : 'bg-slate-100 dark:bg-slate-700 p-3 rounded-lg max-w-[80%]';
            div.innerText = text;
            chatMessages.appendChild(div);
            chatMessages.scrollTop = chatMessages.scrollHeight;
        }

        function handleChat() {
            const msg = chatInput.value.trim();
            if (msg) {
                addMessage(msg, 'user');
                chatInput.value = '';
                
                setTimeout(() => {
                    const responses = [
                        "I can certainly help you schedule an appointment. Would you prefer a morning or afternoon slot?",
                        "That sounds like it might need our doctor's attention. Our next available opening is tomorrow at 10 AM.",
                        "Our clinic is located at 123 Dental Plaza. Would you like a map link?",
                        "Yes, we accept most major dental insurance providers!"
                    ];
                    addMessage(responses[Math.floor(Math.random() * responses.length)], 'bot');
                }, 1000);
            }
        }

        sendChat.addEventListener('click', handleChat);
        chatInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') handleChat();
        });

            setTimeout(() => {
                btn.innerText = originalText;
                btn.classList.replace('bg-green-600', 'bg-teal-600');
                e.target.reset();
            }, 3000);
        });
